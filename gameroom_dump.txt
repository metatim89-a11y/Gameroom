        const paddleWidth = 10, paddleHeight = 80;
        let p1Y = 160, p2Y = 160;
        let ballX = 300, ballY = 200, ballDX = 5, ballDY = 5;

        document.addEventListener('keydown', e => {
            if(e.key === 'ArrowUp') p1Y = Math.max(0, p1Y - 30);
            if(e.key === 'ArrowDown') p1Y = Math.min(canvas.height - paddleHeight, p1Y + 30);
        });

        function resetBall() {
            ballX = canvas.width/2; ballY = canvas.height/2;
            ballDX = -ballDX / Math.abs(ballDX) * 5; ballDY = (Math.random() - 0.5) * 10;
        }

        function update() {
            ballX += ballDX; ballY += ballDY;
            // AI Logic
            const targetY = ballY - paddleHeight/2;
            if(p2Y < targetY) p2Y += 4.5; else p2Y -= 4.5;
            p2Y = Math.max(0, Math.min(canvas.height - paddleHeight, p2Y));

            if(ballY <= 0 || ballY >= canvas.height) ballDY = -ballDY;
            if(ballX <= paddleWidth) {
                if(ballY > p1Y && ballY < p1Y + paddleHeight) {
                    ballDX = -ballDX * 1.05;
                    const deltaY = ballY - (p1Y + paddleHeight/2);
                    ballDY = deltaY * 0.3;
                } else {
                    p2Score++; document.getElementById('p2-score').innerText = p2Score;
                    resetBall();
                }

                    ballDY = deltaY * 0.3;

                    resetBall();
                }
            }
            draw();
        }

            ctx.fillStyle = '#050508'; ctx.fillRect(0,0,canvas.width,canvas.height);

            ctx.shadowBlur = 15; ctx.shadowColor = 'var(--primary)';
            ctx.fillRect(2, p1Y, paddleWidth, paddleHeight);
            ctx.fillRect(canvas.width - paddleWidth - 2, p2Y, paddleWidth, paddleHeight);
            ctx.beginPath(); ctx.arc(ballX, ballY, 8, 0, Math.PI*2); ctx.fill();
            ctx.shadowBlur = 0;
        }

        function resetGame() { p1Score = 0; p2Score = 0; document.getElementById('p1-score').innerText = 0; document.getElementById('p2-score').innerText = 0; resetBall(); }
